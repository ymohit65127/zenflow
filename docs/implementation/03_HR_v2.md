## 1. Overview & Architecture

HR v2 is a complete overhaul of ZenFlow's Human Resources module, expanding from basic employee records to a full HRMS covering:

- **Employee Lifecycle** — Pre-hire → Onboarding → Active → Notice Period → Exit
- **Payroll Engine** — Indian statutory compliance (PF, ESI, PT, TDS), CTC breakdown, payslip PDF generation
- **Leave Management** — Multi-type leaves, accrual engine, multi-level approval workflow
- **Attendance** — Multi-source check-in (biometric, geo, web), regularization, shift management
- **Performance Management** — OKR goals, review cycles, 360-degree feedback, calibration
- **Recruitment** — ATS with pipeline stages, job postings, candidate tracking
- **Onboarding** — Template-based task assignment for multi-department coordination

### Data Flow

```
Employee Created → Salary Assigned → Shift Assigned → Leave Balances Initialized
     ↓                                                           ↓
Attendance Captured (daily) → Monthly Payroll Run → Payslip PDF (MinIO) → Bank Export
     ↓
Leave Requests → Approval Workflow → Balance Deduction → Attendance Update
     ↓
Performance Review Cycle → Self Review → Manager Review → Calibration → Share
```

### 6.1 Payroll Computation Algorithm

```typescript
// lib/hr/payroll-engine.ts

export async function computePayrollEntry(
  employeeId: string,
  periodId: string,
  prisma: PrismaClient
): Promise<HrPayrollEntryComputed> {
  // STEP 1: Fetch employee salary structure
  const salary = await prisma.hrEmployeeSalary.findFirst({
    where: { employee_id: employeeId, effective_to: null },
    include: { structure: true },
  });
  if (!salary) throw new Error("No active salary assignment");

  const ctc = salary.ctc; // Annual CTC
  const components = salary.structure.components as SalaryComponent[];

  // STEP 2: Compute gross monthly salary from CTC
  const grossMonthly = ctc / 12;

  // STEP 3: Resolve each component value
  const resolved: ResolvedComponent[] = [];
  let basic = 0, hra = 0, special = 0, otherAllowances = 0;

  for (const comp of components.sort((a, b) => a.sort_order - b.sort_order)) {
    let value = 0;
    if (comp.type === "fixed") {
      value = comp.value;
    } else if (comp.type === "percentage") {
      // formula_base: "gross" | "basic" | "ctc"
      const base = comp.formula_base === "gross" ? grossMonthly
                 : comp.formula_base === "basic"  ? basic
                 : ctc;
      value = (base * comp.value) / 100;
    } else if (comp.type === "formula") {
      value = evaluateFormula(comp.value, { basic, gross: grossMonthly, ctc });
    }

    if (comp.code === "BASIC") basic = value;
    else if (comp.code === "HRA") hra = value;
    else if (comp.code === "SPECIAL") special = value;
    else otherAllowances += value;

    resolved.push({ ...comp, computed_value: value });
  }

  const grossSalary = basic + hra + special + otherAllowances;

  // STEP 4: Fetch attendance for period
  const period = await prisma.hrPayrollPeriod.findUnique({ where: { id: periodId } });
  const attendance = await prisma.hrAttendance.findMany({
    where: {
      employee_id: employeeId,
      date: { gte: period!.start_date, lte: period!.end_date },
    },
  });

  const workingDays = getWorkingDays(period!.start_date, period!.end_date, [0, 6]);
  const presentDays = attendance.filter(a => ["present", "half_day", "work_from_home"].includes(a.status)).length;
  const halfDays = attendance.filter(a => a.status === "half_day").length;
  const leaveDays = attendance.filter(a => a.status === "on_leave").length;
  const paidLeaveRequests = await getPaidLeaveCount(employeeId, period!.start_date, period!.end_date, prisma);

  const paidDays = presentDays - halfDays * 0.5 + paidLeaveRequests;
  const lopDays = Math.max(0, workingDays - paidDays);

  // STEP 5: Apply LOP
  const lopDeductionFactor = lopDays / workingDays;
  const effectiveGross = grossSalary * (1 - lopDeductionFactor);
  const effectiveBasic = basic * (1 - lopDeductionFactor);

  // STEP 6: Statutory deductions

  // PF (Provident Fund) - 12% of basic, capped at ₹1800/month (15000 ceiling)
  const employee = await prisma.hrEmployee.findUnique({ where: { id: employeeId } });
  let pfEmployee = 0, pfEmployer = 0;
  if (employee?.pf_eligible) {
    const pfBase = Math.min(effectiveBasic, 15000);
    pfEmployee = Math.round(pfBase * 0.12);
    pfEmployer = Math.round(pfBase * 0.12); // Employer contribution (3.67% to EPF + 8.33% to EPS)
  }

  // ESIC - 0.75% employee, 3.25% employer (only if gross <= 21000)
  let esicEmployee = 0, esicEmployer = 0;
  if (employee?.esic_eligible && effectiveGross <= 21000) {
    esicEmployee = Math.round(effectiveGross * 0.0075);
    esicEmployer = Math.round(effectiveGross * 0.0325);
  }

  // Professional Tax (PT) - state-specific slabs
  let pt = 0;
  if (employee?.pt_applicable && employee?.pt_state) {
    pt = getPtSlab(employee.pt_state, effectiveGross);
  }

  // TDS - simplified annual projection method
  const annualGross = effectiveGross * 12;
  const tdsMonthly = computeTdsMonthly(annualGross, pfEmployee * 12);

  const totalDeductions = pfEmployee + esicEmployee + pt + tdsMonthly;
  const netSalary = effectiveGross - totalDeductions;

  return {
    period_id: periodId,
    employee_id: employeeId,
    working_days: workingDays,
    paid_days: paidDays,
    lop_days: lopDays,
    gross_salary: effectiveGross,
    basic: effectiveBasic,
    hra: hra * (1 - lopDeductionFactor),
    special_allowance: special * (1 - lopDeductionFactor),
    other_allowances: otherAllowances * (1 - lopDeductionFactor),
    pf_employee: pfEmployee,
    pf_employer: pfEmployer,
    esic_employee: esicEmployee,
    esic_employer: esicEmployer,
    pt,
    tds: tdsMonthly,
    other_deductions: 0,
    total_deductions: totalDeductions,
    net_salary: netSalary,
    components: resolved,
  };
}

// India Professional Tax slabs by state (monthly gross)
function getPtSlab(state: string, monthlyGross: number): number {
  const slabs: Record<string, [number, number][]> = {
    Maharashtra: [[7500, 0], [10000, 175], [Infinity, 200]],
    Karnataka:   [[15000, 0], [Infinity, 200]],
    Gujarat:     [[5999, 0], [8999, 80], [11999, 150], [Infinity, 200]],
    // ... add all states
  };
  const stateSlab = slabs[state] ?? [];
  for (const [limit, tax] of stateSlab) {
    if (monthlyGross <= limit) return tax;
  }
  return 0;
}

// New Tax Regime FY 2024-25 slabs (simplified)
function computeTdsMonthly(annualGross: number, annualPf: number): number {
  const taxableIncome = annualGross - annualPf - 50000; // standard deduction
  let tax = 0;
  if (taxableIncome <= 300000) tax = 0;
  else if (taxableIncome <= 600000) tax = (taxableIncome - 300000) * 0.05;
  else if (taxableIncome <= 900000) tax = 15000 + (taxableIncome - 600000) * 0.10;
  else if (taxableIncome <= 1200000) tax = 45000 + (taxableIncome - 900000) * 0.15;
  else if (taxableIncome <= 1500000) tax = 90000 + (taxableIncome - 1200000) * 0.20;
  else tax = 150000 + (taxableIncome - 1500000) * 0.30;
  // Rebate under 87A
  if (taxableIncome <= 700000) tax = 0;
  const cess = tax * 0.04;
  return Math.round((tax + cess) / 12);
}
```

### 6.2 Leave Balance Accrual Logic

```typescript
// lib/hr/leave-accrual.ts

export async function accrueMonthlyLeaves(
  orgId: string,
  month: number,
  year: number,
  prisma: PrismaClient
): Promise<void> {
  const leaveTypes = await prisma.hrLeaveType.findMany({
    where: { organization_id: orgId, is_active: true, accrual_type: "monthly" },
  });

  const employees = await prisma.hrEmployee.findMany({
    where: { organization_id: orgId, lifecycle_stage: "active" },
  });

  for (const lt of leaveTypes) {
    for (const emp of employees) {
      // Check min_service_days
      const serviceMonths = differenceInMonths(new Date(), emp.join_date);
      if (serviceMonths * 30 < lt.min_service_days) continue;

      // Check applicable_gender
      if (lt.applicable_gender !== "all") {
        if (emp.gender !== lt.applicable_gender) continue;
      }

      // Upsert balance, add accrual_amount
      await prisma.hrLeaveBalance.upsert({
        where: { employee_id_leave_type_id_year: { employee_id: emp.id, leave_type_id: lt.id, year } },
        create: {
          employee_id: emp.id, leave_type_id: lt.id, year,
          opening_balance: 0, accrued: lt.accrual_amount, taken: 0,
          adjusted: 0, carry_forward: 0, closing_balance: lt.accrual_amount,
        },
        update: {
          accrued: { increment: lt.accrual_amount },
          closing_balance: { increment: lt.accrual_amount },
        },
      });
    }
  }
}

export async function initializeYearlyBalances(
  orgId: string, year: number, prisma: PrismaClient
): Promise<void> {
  const prevYear = year - 1;
  const employees = await prisma.hrEmployee.findMany({ where: { organization_id: orgId, lifecycle_stage: "active" } });
  const leaveTypes = await prisma.hrLeaveType.findMany({ where: { organization_id: orgId } });

  for (const emp of employees) {
    for (const lt of leaveTypes) {
      const prevBalance = await prisma.hrLeaveBalance.findUnique({
        where: { employee_id_leave_type_id_year: { employee_id: emp.id, leave_type_id: lt.id, year: prevYear } },
      });

      let carryForward = 0;
      if (lt.carry_forward_enabled && prevBalance) {
        carryForward = Math.min(prevBalance.closing_balance, lt.carry_forward_max);
      }

      const openingBalance = carryForward + (lt.accrual_type === "yearly" ? lt.accrual_amount : 0);

      await prisma.hrLeaveBalance.create({
        data: {
          employee_id: emp.id, leave_type_id: lt.id, year,
          opening_balance: openingBalance, accrued: 0, taken: 0,
          adjusted: 0, carry_forward: carryForward, closing_balance: openingBalance,
        },
      });
    }
  }
}
```

### 6.3 Org Chart Recursive Query

```typescript
// lib/hr/org-chart.ts — uses PostgreSQL CTE for performance

export async function getOrgChartTree(
  orgId: string,
  rootEmployeeId: string | null,
  prisma: PrismaClient
): Promise<OrgChartNode[]> {
  const rawResult = await prisma.$queryRaw<OrgChartRow[]>`
    WITH RECURSIVE org_tree AS (
      SELECT
        e.id, e.first_name, e.last_name, e.photo_url, e.manager_id,
        d.name as department_name, des.name as designation_name,
        0 as depth
      FROM hr_employees e
      LEFT JOIN hr_departments d ON e.department_id = d.id
      LEFT JOIN hr_designations des ON e.designation_id = des.id
      WHERE e.organization_id = ${orgId}
        AND e.lifecycle_stage = 'active'
        AND e.manager_id IS ${rawNull(rootEmployeeId)}
        ${rootEmployeeId ? Prisma.sql`AND e.id = ${rootEmployeeId}` : Prisma.empty}

      UNION ALL

      SELECT
        e.id, e.first_name, e.last_name, e.photo_url, e.manager_id,
        d.name, des.name, ot.depth + 1
      FROM hr_employees e
      LEFT JOIN hr_departments d ON e.department_id = d.id
      LEFT JOIN hr_designations des ON e.designation_id = des.id
      INNER JOIN org_tree ot ON e.manager_id = ot.id
      WHERE e.organization_id = ${orgId} AND e.lifecycle_stage = 'active'
    )
    SELECT * FROM org_tree ORDER BY depth, last_name
  `;

  return buildTree(rawResult);
}

function buildTree(rows: OrgChartRow[]): OrgChartNode[] {
  const map = new Map<string, OrgChartNode>();
  const roots: OrgChartNode[] = [];

  for (const row of rows) {
    map.set(row.id, { ...row, children: [] });
  }

  for (const row of rows) {
    const node = map.get(row.id)!;
    if (row.manager_id && map.has(row.manager_id)) {
      map.get(row.manager_id)!.children.push(node);
    } else {
      roots.push(node);
    }
  }

  return roots;
}
```

### 6.4 Leave Days Calculation (Excluding Weekends & Holidays)

```typescript
export async function calculateLeaveDays(
  employeeId: string,
  startDate: Date,
  endDate: Date,
  startDayType: HrDayType,
  endDayType: HrDayType,
  prisma: PrismaClient
): Promise<number> {
  // 1. Fetch employee's active shift for weekly off days
  const shiftAssignment = await prisma.hrShiftAssignment.findFirst({
    where: { employee_id: employeeId, effective_from: { lte: startDate }, OR: [{ effective_to: null }, { effective_to: { gte: endDate } }] },
    include: { shift: true },
  });
  const weeklyOffDays: number[] = shiftAssignment?.shift.weekly_off_days ?? [0, 6];

  // 2. Fetch holidays in range (org-level calendar for this year)
  const holidays = await prisma.hrHoliday.findMany({
    where: { calendar: { organization_id: /* orgId */ '', year: startDate.getFullYear() }, date: { gte: startDate, lte: endDate }, is_half_day: false },
  });
  const holidayDates = new Set(holidays.map(h => h.date.toISOString().slice(0, 10)));

  // 3. Iterate days
  let days = 0;
  const cursor = new Date(startDate);
  while (cursor <= endDate) {
    const dayOfWeek = cursor.getDay();
    const dateStr = cursor.toISOString().slice(0, 10);

    if (!weeklyOffDays.includes(dayOfWeek) && !holidayDates.has(dateStr)) {
      // Check half-day
      if (cursor.getTime() === startDate.getTime() && startDayType !== "full") days += 0.5;
      else if (cursor.getTime() === endDate.getTime() && endDayType !== "full") days += 0.5;
      else days += 1;
    }

    cursor.setDate(cursor.getDate() + 1);
  }

  return days;
}
```
